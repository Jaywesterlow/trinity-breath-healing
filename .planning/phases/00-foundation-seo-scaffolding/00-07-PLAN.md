    ```svelte
    <script lang="ts">
      // PRF-01: type-safe wrapper around <enhanced:img>.
      // Hero usage: <EnhancedImage src={...} alt='...' width={W} height={H} loading='eager' fetchpriority='high' />
      // Below-fold usage: <EnhancedImage src={...} alt='...' width={W} height={H} />  (lazy default)

      type Props = {
        src: string;            // path importable via `?enhanced` or plain URL
        alt: string;            // required for accessibility (A11Y-04) AND PRF-01
        width: number;          // required for CLS=0
        height: number;         // required for CLS=0
        loading?: 'eager' | 'lazy';
        fetchpriority?: 'high' | 'low' | 'auto';
        class?: string;
      };

      let {
        src,
        alt,
        width,
        height,
        loading = 'lazy',
        fetchpriority,
        class: className = '',
      }: Props = $props();
    </script>

    {#if fetchpriority}
      <enhanced:img {src} {alt} {width} {height} {loading} {fetchpriority} class={className} />
    {:else}
      <enhanced:img {src} {alt} {width} {height} {loading} class={className} />
    {/if}

    <style>
      enhanced\:img, img {
        display: block;
        max-width: 100%;
        height: auto;
      }
    </style>
    ```
